\begin{align}
    \frac{V_{-} - 0.1}{R_1} + C_1\frac{d \brak{V_{-} - V_{out}}}{dt} & = 0\\
    \frac{d \brak{V_{out}}}{dt} & = -\frac{1}{R_1C_1}\\
    V_{out} & = -\frac{1}{R_1C_1}t\\
    V_{out} & = max\cbrak{-V_{EE}, -\frac{1}{R_1C_1}t}\\
\end{align}

for $t>V_{EE}R_1C_1$, $V_{out}$ saturates to $-V_{EE}$
